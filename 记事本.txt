周报数据SQL：
SELECT 
  country,
  SUM(imp_count) / COUNT(DISTINCT date) AS avg_imp_per_day
FROM (
  SELECT 
    country,
    TIMESTAMP_TRUNC(_PARTITIONTIME, DAY) AS date,
    COUNT(*) AS imp_count
  FROM `feed-435306.feed_stats.feed_event_p`
  WHERE 
    _PARTITIONTIME BETWEEN TIMESTAMP("2025-06-30") AND TIMESTAMP("2025-07-06")
    AND imp = 1
  GROUP BY country, date
)
GROUP BY country
ORDER BY avg_imp_per_day DESC
